table User {
    username varchar(30) <pk>
}

table TaskList {
    title varchar(100) <pk>
    owner varchar(30) <pk,fk>
    description varchar(1000)
}

table Task {
    title varchar(100) <pk>
    list_title varchar(100) <pk,fk1>
    owner varchar(30) <pk,fk2>
    description varchar(1000)
    start_date date
    end_date date
    start_time time
    end_time time
}
